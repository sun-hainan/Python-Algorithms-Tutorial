## 第一部分：什么是回溯算法？

### 🎯 核心思想

**一条路走到黑，发现走不通就回退，换一条路继续走。**

```
回溯 = 递归 + 剪枝

尝试 → 失败 → 回退 → 尝试另一条路
```

## 第二部分：八皇后问题

### 八皇后代码

```python
def solve_n_queens(n=8):
    solutions = []
    board = [-1] * n

    def is_valid(board, row, col):
        for prev_row in range(row):
            prev_col = board[prev_row]
            if prev_col == col or abs(prev_col - col) == abs(prev_row - row):
                return False
        return True

    def backtrack(row):
        if row == n:
            solutions.append(board[:])
            return
        for col in range(n):
            if is_valid(board, row, col):
                board[row] = col
                backtrack(row + 1)
                board[row] = -1

    backtrack(0)
    return solutions

print(f"8皇后共有 {len(solve_n_queens(8))} 个解")  # 92
```

## 第三部分：迷宫问题

```python
def solve_maze(maze, start, end):
    rows, cols = len(maze), len(maze[0])
    visited = set()
    path = []

    def dfs(x, y):
        if (x, y) == end:
            path.append((x, y))
            return True
        visited.add((x, y))
        path.append((x, y))
        for dx, dy in [(0,1),(0,-1),(1,0),(-1,0)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < rows and 0 <= ny < cols and maze[nx][ny] == 0 and (nx, ny) not in visited:
                if dfs(nx, ny):
                    return True
        path.pop()
        return False

    dfs(start[0], start[1])
    return path
```

## 第四部分：全排列

```python
def permutations(nums):
    result = []
    used = [False] * len(nums)
    path = []

    def backtrack():
        if len(path) == len(nums):
            result.append(path[:])
            return
        for i in range(len(nums)):
            if not used[i]:
                used[i] = True
                path.append(nums[i])
                backtrack()
                path.pop()
                used[i] = False

    backtrack()
    return result
```
