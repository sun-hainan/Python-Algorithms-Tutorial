## 🤔 什么是堆？

### 堆的定义

```
堆（Heap）是一种完全二叉树，满足堆性质：

最大堆（Max Heap）：
  - 父节点 >= 子节点
  - 根节点是最大值

最小堆（Min Heap）：
  - 父节点 <= 子节点
  - 根节点是最小值
```

### 最大堆示例

```
最大堆：
           90
         /    \
       80      70
      /  \    /  \
    50   60  55   65
   / \
  30  40

特点：
- 根节点 90 是最大值
- 每个父节点都 >= 它的子节点
```

### 最小堆示例

```
最小堆：
           10
         /    \
       30      20
      /  \    /  \
    90   50  40   55

特点：
- 根节点 10 是最小值
- 每个父节点都 <= 它的子节点
```

## 📐 堆的存储方式

### 完全二叉树的数组表示

```
完全二叉树：
           0
         /   \
        1     2
       / \   /
      3   4 5

数组表示：[0, 1, 2, 3, 4, 5]

关键公式（从索引 0 开始）：
- 父节点索引 = (i - 1) // 2
- 左孩子索引 = 2 * i + 1
- 右孩子索引 = 2 * i + 2

验证：
- 节点 1：父 = (1-1)//2 = 0 ✓
- 节点 2：父 = (2-1)//2 = 0 ✓
- 节点 3：父 = (3-1)//2 = 1 ✓
```

### 数组存储的优点

```
- 不需要指针，节省内存
- 可以用数学公式快速访问父节点和孩子
- 完全二叉树没有"洞"，内存利用率高
```

## 💻 堆的代码实现

### 最小堆

```python
class MinHeap:
    """最小堆"""
    def __init__(self):
        self.heap = []

    # ============ 辅助函数 ============
    def _parent(self, i):
        """父节点索引"""
        return (i - 1) // 2

    def _left(self, i):
        """左孩子索引"""
        return 2 * i + 1

    def _right(self, i):
        """右孩子索引"""
        return 2 * i + 2

    def _swap(self, i, j):
        """交换两个元素"""
        self.heap[i], self.heap[j] = self.heap[j], self.heap[i]

    def _sift_up(self, i):
        """
        上浮操作
        把节点向上移动到正确位置
        用于插入后调整
        """
        while i > 0:
            parent = self._parent(i)
            if self.heap[i] < self.heap[parent]:
                self._swap(i, parent)
                i = parent
            else:
                break

    def _sift_down(self, i):
        """
        下沉操作
        把节点向下移动到正确位置
        用于删除后调整
        """
        n = len(self.heap)
        while True:
            smallest = i
            left = self._left(i)
            right = self._right(i)

            if left < n and self.heap[left] < self.heap[smallest]:
                smallest = left
            if right < n and self.heap[right] < self.heap[smallest]:
                smallest = right

            if smallest != i:
                self._swap(i, smallest)
                i = smallest
            else:
                break

    # ============ 主要操作 ============
    def push(self, val):
        """插入元素 O(log n)"""
        self.heap.append(val)
        self._sift_up(len(self.heap) - 1)

    def pop(self):
        """
        弹出最小元素 O(log n)
        1. 保存根（最小值）
        2. 把最后一个元素移到根
        3. 下沉调整
        """
        if not self.heap:
            raise IndexError("堆为空")

        min_val = self.heap[0]

        last = self.heap.pop()
        if self.heap:
            self.heap[0] = last
            self._sift_down(0)

        return min_val

    def peek(self):
        """查看最小元素 O(1)"""
        if not self.heap:
            raise IndexError("堆为空")
        return self.heap[0]

    def size(self):
        return len(self.heap)

    def is_empty(self):
        return len(self.heap) == 0

    def __str__(self):
        return str(self.heap)


# 测试
heap = MinHeap()

# 插入
for val in [50, 30, 40, 10, 60, 20]:
    heap.push(val)
    print(f"插入 {val}: {heap}")

# 弹出
print("\n弹出过程：")
while not heap.is_empty():
    print(f"弹出 {heap.pop()}: 剩余 {heap}")
```

## 📐 堆排序（Heap Sort）

### 堆排序思想

```
1. 把数组建成最大堆
2. 交换堆顶（最大值）和最后一个元素
3. 把剩余 n-1 个元素重新堆化
4. 重复 2-3，直到只剩一个元素
```

### 堆排序代码

```python
def heap_sort(arr):
    """
    堆排序 O(n log n)
    """
    n = len(arr)

    # ============ 第一步：建成最大堆 ============
    def sift_down(start, end):
        """下沉操作"""
        while True:
            parent = start
            child = 2 * start + 1

            if child > end:
                break

            # 找最大的孩子
            if child + 1 <= end and arr[child] < arr[child + 1]:
                child += 1
            if child > end:
                break

            # 交换并继续下沉
            if arr[child] > arr[parent]:
                arr[parent], arr[child] = arr[child], arr[parent]
                parent = child
                child = 2 * parent + 1
            else:
                break

    # 从最后一个父节点开始，向前下沉
    # 最后一个节点的父节点是 (n-2)//2
    for start in range((n - 2) // 2, -1, -1):
        sift_down(start, n - 1)

    # ============ 第二步：堆排序 ============
    for end in range(n - 1, 0, -1):
        # 把最大的放到最后
        arr[0], arr[end] = arr[end], arr[0]
        # 把剩余的重新堆化
        sift_down(0, end - 1)

    return arr


# 测试
arr = [50, 30, 40, 10, 60, 20]
print(f"原始数组: {arr}")
heap_sort(arr)
print(f"排序后:   {arr}")
```

## 📐 优先级队列（用堆实现）

### 优先级队列 vs 普通队列

```
普通队列（FIFO）：      优先级队列：
┌─────────────────┐    ┌─────────────────┐
│  任务1 (先来)    │    │  任务3 (最紧急)  │
│  任务2           │ →  │  任务1 (次紧急)  │
│  任务3 (后来)    │    │  任务2 (不紧急)  │
└─────────────────┘    └─────────────────┘
```

### 优先级队列实现

```python
import heapq  # Python 内置堆

class PriorityQueue:
    """优先级队列（最小堆实现）"""
    def __init__(self):
        self.heap = []

    def enqueue(self, item, priority=0):
        """入队 O(log n)"""
        # Python heapq 是最小堆，用 (priority, item) 元组
        heapq.heappush(self.heap, (priority, item))

    def dequeue(self):
        """出队 O(log n)"""
        if self.is_empty():
            raise IndexError("队列为空")
        return heapq.heappop(self.heap)[1]

    def peek(self):
        """查看最高优先级 O(1)"""
        if self.is_empty():
            raise IndexError("队列为空")
        return self.heap[0][1]

    def is_empty(self):
        return len(self.heap) == 0

    def size(self):
        return len(self.heap)


# 测试
pq = PriorityQueue()
pq.enqueue("任务A", 3)  # 普通
pq.enqueue("任务B", 1)  # 紧急
pq.enqueue("任务C", 2)  # 比较紧急
pq.enqueue("任务D", 5)  # 不紧急

print("入队顺序: A(3), B(1), C(2), D(5)")
print(f"出队顺序: {pq.dequeue()}")  # B (最紧急)
print(f"出队顺序: {pq.dequeue()}")  # C
print(f"出队顺序: {pq.dequeue()}")  # A
print(f"出队顺序: {pq.dequeue()}")  # D
```

## 🧪 经典面试题

### 题目1：找第 K 大的元素

```python
import heapq

def find_kth_largest(nums, k):
    """
    找第 K 大的元素
    方法1：排序 O(n log n)
    方法2：堆 O(n log k)
    """
    # 用最小堆，保持 k 个最大元素
    heap = []

    for num in nums:
        heapq.heappush(heap, num)
        if len(heap) > k:
            heapq.heappop(heap)

    return heap[0]


# 测试
nums = [3, 2, 1, 5, 6, 4]
print(f"第2大的元素: {find_kth_largest(nums, 2)}")  # 5
print(f"第3大的元素: {find_kth_largest(nums, 3)}")  # 4
```

### 题目2：合并 K 个有序数组

```python
import heapq

def merge_k_sorted_arrays(arrays):
    """合并 K 个有序数组"""
    heap = []

    # 初始化：把每个数组的第一个元素放入堆
    for i, arr in enumerate(arrays):
        if arr:
            heapq.heappush(heap, (arr[0], i, 0))

    result = []

    while heap:
        val, arr_idx, elem_idx = heapq.heappop(heap)
        result.append(val)

        # 把该数组的下一个元素放入堆
        if elem_idx + 1 < len(arrays[arr_idx]):
            next_val = arrays[arr_idx][elem_idx + 1]
            heapq.heappush(heap, (next_val, arr_idx, elem_idx + 1))

    return result


# 测试
arrays = [
    [1, 4, 7],
    [2, 5, 8],
    [3, 6, 9]
]
print(merge_k_sorted_arrays(arrays))  # [1, 2, 3, 4, 5, 6, 7, 8, 9]
```

### 题目3：滑动窗口最大值

```python
from collections import deque

def sliding_window_maximum(nums, k):
    """
    滑动窗口最大值
    方法：双端队列（单调递减队列）

    例如：[1,3,-1,-3,5,3,6,7], k=3
    结果：[3,3,5,5,6,7]
    """
    if not nums or k == 0:
        return []

    deq = deque()  # 存储索引
    result = []

    for i, num in enumerate(nums):
        # 移除不在窗口内的元素
        while deq and deq[0] <= i - k:
            deq.popleft()

        # 保持递减：移除所有比当前元素小的
        while deq and nums[deq[-1]] < num:
            deq.pop()

        # 添加当前元素
        deq.append(i)

        # 记录窗口最大值
        if i >= k - 1:
            result.append(nums[deq[0]])

    return result


# 测试
nums = [1, 3, -1, -3, 5, 3, 6, 7]
k = 3
print(sliding_window_maximum(nums, k))  # [3, 3, 5, 5, 6, 7]
```

## 📊 堆的时间复杂度

| 操作 | 时间复杂度 | 说明 |
|------|----------|------|
| push（插入）| O(log n) | 上浮调整 |
| pop（弹出）| O(log n) | 下沉调整 |
| peek（查看）| O(1) | 直接返回根 |
| 建堆 | O(n) | 从后往前下沉 |
