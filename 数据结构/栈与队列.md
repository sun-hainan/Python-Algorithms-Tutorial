# 栈与队列

> _栈是弹匣，队列是排队——两种截然不同的生活哲学_

---

## 🎯 先看一个生活中的例子

### 栈：弹匣（Last In, First Out）

```
弹匣的子弹：
┌─────┐
│ 3号 │ ← 最后装入，最先射出
│ 2号 │
│ 1号 │
│ 4号 │ ← 最先装入，最后射出
└─────┘

LIFO = Last In First Out（后进先出）
```

### 队列：排队买票（FIFO）

```
排队买票：
┌─────┬─────┬─────┬─────┐
│ 小明 │ 小红 │ 小刚 │ 小李 │ ...
│ 队首 │      │      │ 队尾
└─────┴─────┴─────┴─────┘

FIFO = First In First Out（先进先出）
```

## 📐 栈详解

### 栈的基本操作

```
┌─────────────────────────────────────┐
│              栈 (Stack)              │
├─────────────────────────────────────┤
│  push(x)  │ 压栈：把 x 放到栈顶     │
│  pop()    │ 弹栈：移除并返回栈顶    │
│  peek()   │ 查看栈顶（不删除）      │
│  is_empty │ 判断栈是否为空          │
└─────────────────────────────────────┘

示意图：

    push(4)     push(5)     pop()      push(6)
    ┌───┐      ┌───┐      ┌───┐      ┌───┐
    │ 4 │      │ 5 │      │ 4 │      │ 6 │
    └───┘      ├───┤      ├───┤      ├───┤
               │ 4 │      │   │      │ 4 │
               └───┘      └───┘      └───┘
```

### 栈的代码实现（用数组）

```python
class Stack:
    """栈 LIFO - 用数组实现"""
    def __init__(self):
        self.items = []

    def push(self, item):
        """压栈 O(1)"""
        self.items.append(item)

    def pop(self):
        """弹栈 O(1)"""
        if self.is_empty():
            raise IndexError("栈为空")
        return self.items.pop()

    def peek(self):
        """查看栈顶 O(1)"""
        if self.is_empty():
            raise IndexError("栈为空")
        return self.items[-1]

    def is_empty(self):
        """判断是否为空 O(1)"""
        return len(self.items) == 0

    def size(self):
        return len(self.items)


# 测试
stack = Stack()
stack.push(10)
stack.push(20)
stack.push(30)
print(stack.peek())   # 30
print(stack.pop())    # 30
print(stack.pop())    # 20
print(stack.is_empty())  # False
```

### 栈的应用场景

#### 1. 括号匹配

```python
def is_valid(s):
    """
    判断括号字符串是否有效
    例如："()" 有效，"()[]{}" 有效，"(]" 无效
    """
    stack = Stack()
    pairs = {')': '(', ']': '[', '}': '{'}

    for char in s:
        if char in '([{':
            stack.push(char)
        elif char in ')]}':
            if stack.is_empty():
                return False
            if stack.pop() != pairs[char]:
                return False

    return stack.is_empty()


# 测试
print(is_valid("()"))       # True
print(is_valid("()[]{}"))   # True
print(is_valid("(]"))       # False
print(is_valid("([)]"))      # False
```

#### 2. 表达式求值（后缀表达式）

```python
def evaluate_postfix(expr):
    """
    计算后缀表达式
    例如："3 4 + 2 *" = (3 + 4) * 2 = 14
    """
    stack = Stack()
    tokens = expr.split()

    for token in tokens:
        if token.isdigit():
            stack.push(int(token))
        else:
            b = stack.pop()
            a = stack.pop()
            if token == '+':
                stack.push(a + b)
            elif token == '-':
                stack.push(a - b)
            elif token == '*':
                stack.push(a * b)
            elif token == '/':
                stack.push(int(a / b))

    return stack.pop()


# 测试
print(evaluate_postfix("3 4 + 2 *"))  # 14
print(evaluate_postfix("15 7 1 1 + - / 3 * 2 1 + 8 +"))  # 11
```

#### 3. 深度优先搜索（DFS）

```python
def dfs(graph, start):
    """
    用栈实现深度优先搜索
    graph: 邻接表
    """
    visited = set()
    stack = Stack()
    stack.push(start)
    result = []

    while not stack.is_empty():
        node = stack.pop()
        if node not in visited:
            visited.add(node)
            result.append(node)

            # 把邻居压栈（逆序保证顺序）
            for neighbor in reversed(graph[node]):
                if neighbor not in visited:
                    stack.push(neighbor)

    return result


# 测试
graph = {
    'A': ['B', 'C'],
    'B': ['D', 'E'],
    'C': ['F'],
    'D': [],
    'E': ['F'],
    'F': []
}
print(dfs(graph, 'A'))  # ['A', 'C', 'F', 'B', 'E', 'D']
```

## 📐 队列详解

### 队列的基本操作

```
┌─────────────────────────────────────┐
│            队列 (Queue)              │
├─────────────────────────────────────┤
│  enqueue(x) │ 入队：添加到队尾       │
│  dequeue()  │ 出队：移除并返回队首   │
│  front()    │ 查看队首（不删除）     │
│  is_empty   │ 判断队列是否为空       │
└─────────────────────────────────────┘

示意图：

    enqueue(1)  enqueue(2)  enqueue(3)  dequeue()   enqueue(4)
    ┌───┐
    │ 1 │ ← 队首
    └───┘

    ┌───┬───┐
    │ 1 │ 2 │ ← 队首
    └───┴───┘

    ┌───┬───┬───┐
    │ 1 │ 2 │ 3 │ ← 队首
    └───┴───┴───┘

    ┌───┬───┐
    │ 2 │ 3 │ ← 队首
    └───┴───┘

    ┌───┬───┬───┐
    │ 2 │ 3 │ 4 │ ← 队首
    └───┴───┴───┘
```

### 队列的代码实现（数组）

```python
class Queue:
    """队列 FIFO - 用数组实现"""
    def __init__(self):
        self.items = []

    def enqueue(self, item):
        """入队 O(1)"""
        self.items.append(item)

    def dequeue(self):
        """出队 O(n) —— 因为要移动所有元素"""
        if self.is_empty():
            raise IndexError("队列为空")
        return self.items.pop(0)  # 数组头部删除效率低！

    def front(self):
        """查看队首 O(1)"""
        if self.is_empty():
            raise IndexError("队列为空")
        return self.items[0]

    def is_empty(self):
        return len(self.items) == 0

    def size(self):
        return len(self.items)
```

### 循环队列（高效实现）

```python
class CircularQueue:
    """循环队列 - O(1) 入队和出队"""
    def __init__(self, capacity):
        self.capacity = capacity + 1  # 多一个位置区分空和满
        self.items = [None] * self.capacity
        self.front = 0   # 队首索引
        self.rear = 0    # 队尾索引（下一个入队位置）

    def enqueue(self, item):
        """入队 O(1)"""
        if self.is_full():
            raise OverflowError("队列已满")
        self.items[self.rear] = item
        self.rear = (self.rear + 1) % self.capacity

    def dequeue(self):
        """出队 O(1)"""
        if self.is_empty():
            raise IndexError("队列为空")
        item = self.items[self.front]
        self.front = (self.front + 1) % self.capacity
        return item

    def front_item(self):
        """查看队首"""
        if self.is_empty():
            raise IndexError("队列为空")
        return self.items[self.front]

    def is_empty(self):
        return self.front == self.rear

    def is_full(self):
        return (self.rear + 1) % self.capacity == self.front

    def size(self):
        """队列中的元素数量"""
        if self.rear >= self.front:
            return self.rear - self.front
        else:
            return self.capacity - self.front + self.rear


# 测试
q = CircularQueue(5)
for i in range(1, 6):
    q.enqueue(i)
print(q.dequeue())   # 1
print(q.dequeue())   # 2
q.enqueue(6)
print(q.dequeue())   # 3
```

### 队列的应用场景

#### 1. 广度优先搜索（BFS）

```python
def bfs(graph, start):
    """
    用队列实现广度优先搜索
    """
    visited = set()
    queue = Queue()
    queue.enqueue(start)
    visited.add(start)
    result = []

    while not queue.is_empty():
        node = queue.dequeue()
        result.append(node)

        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.enqueue(neighbor)

    return result


# 测试
graph = {
    'A': ['B', 'C'],
    'B': ['D', 'E'],
    'C': ['F'],
    'D': [],
    'E': ['F'],
    'F': []
}
print(bfs(graph, 'A'))  # ['A', 'B', 'C', 'D', 'E', 'F']
```

#### 2. 打印任务队列

```python
class Printer:
    """模拟打印机队列"""
    def __init__(self):
        self.queue = Queue()
        self.current = None

    def add_job(self, document):
        """添加打印任务"""
        self.queue.enqueue(document)

    def print_next(self):
        """打印下一个任务"""
        if self.current is None:
            if self.queue.is_empty():
                return None
            self.current = self.queue.dequeue()
            return self.current
        return None  # 正在打印

    def is_busy(self):
        return self.current is not None
```

#### 3. 循环调度（Round Robin）

```python
def round_robin(tasks, quantum):
    """
    时间片轮转调度
    tasks: 任务列表
    quantum: 时间片长度
    """
    queue = Queue()
    for task in tasks:
        queue.enqueue(task)

    time = 0
    while not queue.is_empty():
        task = queue.dequeue()
        if task['time'] <= quantum:
            time += task['time']
            print(f"任务 {task['name']} 执行 {task['time']} 单位时间，完成于 {time}")
        else:
            time += quantum
            task['time'] -= quantum
            print(f"任务 {task['name']} 执行 {quantum} 单位时间，剩余 {task['time']}")
            queue.enqueue(task)

    return time


# 测试
tasks = [
    {'name': 'A', 'time': 10},
    {'name': 'B', 'time': 5},
    {'name': 'C', 'time': 8}
]
round_robin(tasks, quantum=3)
```

## 📊 栈 vs 队列对比

| 特性 | 栈 | 队列 |
|------|-----|------|
| 简称 | LIFO | FIFO |
| 入口/出口 | 同一端 | 两端 |
| 操作位置 | 栈顶 | 队首/队尾 |
| 应用 | 括号匹配、DFS、递归 | BFS、任务调度 |

## 💻 双端队列（Deque）

```python
class Deque:
    """双端队列，两端都可以入队和出队"""
    def __init__(self):
        self.items = []

    def add_front(self, item):
        """队首添加 O(n)"""
        self.items.insert(0, item)

    def add_rear(self, item):
        """队尾添加 O(1)"""
        self.items.append(item)

    def remove_front(self):
        """队首移除 O(1)"""
        if self.is_empty():
            raise IndexError("队列为空")
        return self.items.pop(0)

    def remove_rear(self):
        """队尾移除 O(1)"""
        if self.is_empty():
            raise IndexError("队列为空")
        return self.items.pop()

    def is_empty(self):
        return len(self.items) == 0

    def size(self):
        return len(self.items)


# Python 内置的 deque
from collections import deque

dq = deque()
dq.append(1)       # 队尾添加
dq.appendleft(0)   # 队首添加 O(1)
dq.pop()          # 队尾移除
dq.popleft()      # 队首移除 O(1)
```
