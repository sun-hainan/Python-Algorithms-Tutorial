# 数组与链表

> _两种最基础的数据结构，一个像整齐的书架，一个像手拉手的朋友_

---

## 🎯 先看一个生活中的例子

### 数组：整齐的书架

```
书架上的书（数组）：
┌─────┬─────┬─────┬─────┬─────┬─────┐
│ 索引0 │ 索引1 │ 索引2 │ 索引3 │ 索引4 │ 索引5 │
│  A书  │  B书  │  C书  │  D书  │  E书  │  F书  │
└─────┴─────┴─────┴─────┴─────┴─────┘

特点：
- 每个位置都有固定编号（索引）
- 知道编号，就能直接找到书 O(1)
- 书架是连续的
```

### 链表：手拉手的朋友

```
小朋友们手拉手（链表）：
[A] → [B] → [C] → [D] → [E] → [F] → NULL

每个小朋友（节点）知道：
- 自己是谁（数据）
- 下一个小朋友是谁（指针）

特点：
- 不用知道编号，顺着找就行
- 不能跳着找，必须从头开始 O(n)
- 不用连续的物理位置
```

## 📐 数组详解

### 数组的内存结构

```
数组名: arr
地址:   0x1000
元素:   [10, 20, 30, 40, 50]

内存示意图：

地址        内容
0x1000  →  10  (arr[0])
0x1004  →  20  (arr[1])   ← 每个 int 占 4 字节
0x1008  →  30  (arr[2])   ← 连续存放！
0x100C  →  40  (arr[3])
0x1010  →  50  (arr[4])

计算 arr[i] 的地址 = 首地址 + i × 每个元素的大小
```

### 数组的操作复杂度

| 操作 | 复杂度 | 说明 |
|------|--------|------|
| 访问 arr[i] | O(1) | 随机访问，直接计算地址 |
| 查找元素 | O(n) | 遍历查找 |
| 在末尾插入 | O(1) | 已知长度时 |
| 在中间插入 | O(n) | 需要移动元素 |
| 删除元素 | O(n) | 需要移动元素 |

### 数组的代码实现

```python
class Array:
    """简单数组实现"""
    def __init__(self, capacity=10):
        self.capacity = capacity
        self.size = 0
        self.data = [None] * capacity

    def get(self, index):
        """获取指定索引的元素 O(1)"""
        if index < 0 or index >= self.size:
            raise IndexError("索引越界")
        return self.data[index]

    def set(self, index, value):
        """设置指定索引的元素 O(1)"""
        if index < 0 or index >= self.size:
            raise IndexError("索引越界")
        self.data[index] = value

    def insert(self, index, value):
        """在指定位置插入元素 O(n)"""
        if self.size >= self.capacity:
            self._resize(self.capacity * 2)

        if index < 0 or index > self.size:
            raise IndexError("索引越界")

        # 从后往前移动元素
        for i in range(self.size, index, -1):
            self.data[i] = self.data[i - 1]

        self.data[index] = value
        self.size += 1

    def delete(self, index):
        """删除指定位置的元素 O(n)"""
        if index < 0 or index >= self.size:
            raise IndexError("索引越界")

        # 从前往后移动元素
        for i in range(index, self.size - 1):
            self.data[i] = self.data[i + 1]

        self.size -= 1

        # 缩容（可选）
        if self.size < self.capacity // 4:
            self._resize(self.capacity // 2)

    def append(self, value):
        """在末尾添加元素 O(1)"""
        if self.size >= self.capacity:
            self._resize(self.capacity * 2)
        self.data[self.size] = value
        self.size += 1

    def _resize(self, new_capacity):
        """动态扩容"""
        new_data = [None] * new_capacity
        for i in range(self.size):
            new_data[i] = self.data[i]
        self.data = new_data
        self.capacity = new_capacity

    def __len__(self):
        return self.size

    def __str__(self):
        return str([self.data[i] for i in range(self.size)])


# 测试
arr = Array(5)
arr.append(10)
arr.append(20)
arr.append(30)
arr.insert(1, 15)
print(arr)        # [10, 15, 20, 30]
arr.delete(2)
print(arr)        # [10, 15, 30]
```

## 📐 链表详解

### 单向链表

```
结构：
┌─────┬────┐    ┌─────┬────┐    ┌─────┬────┐    ┌──────┐
│ 数据  │Next│ →  │ 数据  │Next│ →  │ 数据  │Next│ → NULL
│  10  │    │    │  20  │    │    │  30  │    │
└─────┴────┘    └─────┴────┘    └─────┴────┘    └──────┘
   head
```

### 单向链表的代码实现

```python
class Node:
    """链表节点"""
    def __init__(self, data):
        self.data = data
        self.next = None


class LinkedList:
    """单向链表"""
    def __init__(self):
        self.head = None
        self.size = 0

    def get(self, index):
        """获取指定索引的元素 O(n)"""
        if index < 0 or index >= self.size:
            raise IndexError("索引越界")

        current = self.head
        for _ in range(index):
            current = current.next
        return current.data

    def insert(self, index, value):
        """在指定位置插入元素 O(n)"""
        if index < 0 or index > self.size:
            raise IndexError("索引越界")

        new_node = Node(value)

        if index == 0:
            # 插入头部
            new_node.next = self.head
            self.head = new_node
        else:
            # 找到前一个节点
            prev = self._get_prev(index)
            new_node.next = prev.next
            prev.next = new_node

        self.size += 1

    def _get_prev(self, index):
        """获取指定索引的前一个节点"""
        current = self.head
        for _ in range(index - 1):
            current = current.next
        return current

    def delete(self, index):
        """删除指定位置的元素 O(n)"""
        if index < 0 or index >= self.size:
            raise IndexError("索引越界")

        if index == 0:
            self.head = self.head.next
        else:
            prev = self._get_prev(index)
            prev.next = prev.next.next

        self.size -= 1

    def append(self, value):
        """在末尾添加元素 O(n)"""
        new_node = Node(value)

        if self.head is None:
            self.head = new_node
        else:
            current = self.head
            while current.next:
                current = current.next
            current.next = new_node

        self.size += 1

    def __str__(self):
        """打印链表"""
        result = []
        current = self.head
        while current:
            result.append(str(current.data))
            current = current.next
        return " -> ".join(result) + " -> NULL"

    def __len__(self):
        return self.size


# 测试
ll = LinkedList()
ll.append(10)
ll.append(20)
ll.append(30)
ll.insert(1, 15)
print(ll)         # 10 -> 15 -> 20 -> 30 -> NULL
ll.delete(2)
print(ll)         # 10 -> 15 -> 30 -> NULL
print(ll.get(1))  # 15
```

### 双向链表

```
结构：
NULL  ←  ←  ←  ←  ←  ←  ←  ←  ←  ←
      ┌────┬──────┬────┐
NULL  │Prev│ 数据  │Next│  →  →  →  →  →  NULL
      │    │  20  │    │
      └────┴──────┴────┘
```

### 双向链表的代码实现

```python
class DNode:
    """双向链表节点"""
    def __init__(self, data):
        self.data = data
        self.prev = None
        self.next = None


class DoublyLinkedList:
    """双向链表"""
    def __init__(self):
        self.head = None
        self.tail = None
        self.size = 0

    def append(self, value):
        """在末尾添加 O(1)"""
        new_node = DNode(value)

        if self.tail is None:
            self.head = self.tail = new_node
        else:
            new_node.prev = self.tail
            self.tail.next = new_node
            self.tail = new_node

        self.size += 1

    def prepend(self, value):
        """在头部添加 O(1)"""
        new_node = DNode(value)

        if self.head is None:
            self.head = self.tail = new_node
        else:
            new_node.next = self.head
            self.head.prev = new_node
            self.head = new_node

        self.size += 1

    def delete(self, node):
        """删除指定节点 O(1)"""
        if node.prev:
            node.prev.next = node.next
        else:
            self.head = node.next

        if node.next:
            node.next.prev = node.prev
        else:
            self.tail = node.prev

        self.size -= 1

    def __str__(self):
        result = []
        current = self.head
        while current:
            result.append(str(current.data))
            current = current.next
        return " <-> ".join(result)
```

## 📊 数组 vs 链表对比

| 特性 | 数组 | 链表 |
|------|------|------|
| 内存布局 | 连续 | 不连续 |
| 随机访问 | O(1) | O(n) |
| 头部插入 | O(n) | O(1) |
| 尾部插入 | O(1)* | O(1)* |
| 中间插入 | O(n) | O(n) |
| 删除操作 | O(n) | O(1)** |
| 内存效率 | 高（无额外开销）| 低（需要存指针）|
| 缓存友好性 | 高（连续内存）| 低（可能 cache miss）|

*需要维护长度，或 O(1) 访问到尾部
**需要 O(n) 找到前驱节点

## 🔍 何时用数组？何时用链表？

### 用数组的场景

```
✅ 需要频繁随机访问元素
✅ 数据量固定，不需要频繁插入/删除
✅ 需要良好的缓存命中
✅ 存储简单的数据类型（int、float）
```

### 用链表的场景

```
✅ 需要频繁在头部/中间插入删除
✅ 数据量不确定，可能动态增长
✅ 不需要随机访问，只需要顺序遍历
✅ 需要实现队列、栈等数据结构
```

## 💻 实际应用

### 用链表实现栈

```python
class Stack:
    """用链表实现栈 LIFO"""
    def __init__(self):
        self.top = None
        self.size = 0

    def push(self, value):
        """压栈 O(1)"""
        node = Node(value)
        node.next = self.top
        self.top = node
        self.size += 1

    def pop(self):
        """弹栈 O(1)"""
        if self.top is None:
            raise IndexError("栈为空")
        value = self.top.data
        self.top = self.top.next
        self.size -= 1
        return value

    def peek(self):
        """查看栈顶 O(1)"""
        if self.top is None:
            raise IndexError("栈为空")
        return self.top.data

    def is_empty(self):
        return self.top is None


# 测试
stack = Stack()
stack.push(10)
stack.push(20)
stack.push(30)
print(stack.pop())  # 30
print(stack.pop())  # 20
print(stack.peek()) # 10
```

### 用链表实现队列

```python
class Queue:
    """用链表实现队列 FIFO"""
    def __init__(self):
        self.head = None
        self.tail = None
        self.size = 0

    def enqueue(self, value):
        """入队 O(1)"""
        node = Node(value)
        if self.tail:
            self.tail.next = node
            self.tail = node
        else:
            self.head = self.tail = node
        self.size += 1

    def dequeue(self):
        """出队 O(1)"""
        if self.head is None:
            raise IndexError("队列为空")
        value = self.head.data
        self.head = self.head.next
        if self.head is None:
            self.tail = None
        self.size -= 1
        return value


# 测试
queue = Queue()
queue.enqueue(10)
queue.enqueue(20)
queue.enqueue(30)
print(queue.dequeue())  # 10
print(queue.dequeue())  # 20
```

## 🔗 相关面试题

### 题目1：反转链表

```python
def reverse_linked_list(head):
    """反转单向链表"""
    prev = None
    current = head

    while current:
        next_node = current.next  # 保存下一个
        current.next = prev       # 反转指针
        prev = current             # 移动 prev
        current = next_node        # 移动 current

    return prev  # 新的头节点


# 测试
ll = LinkedList()
for i in [1, 2, 3, 4, 5]:
    ll.append(i)
print(ll)  # 1 -> 2 -> 3 -> 4 -> 5 -> NULL

# 反转
ll.head = reverse_linked_list(ll.head)
print(ll)  # 5 -> 4 -> 3 -> 2 -> 1 -> NULL
```

### 题目2：检测链表是否有环

```python
def has_cycle(head):
    """快慢指针检测环 O(n)"""
    slow = fast = head

    while fast and fast.next:
        slow = slow.next        # 慢指针走一步
        fast = fast.next.next   # 快指针走两步

        if slow == fast:
            return True  # 相遇，有环

    return False  # 无环
```
