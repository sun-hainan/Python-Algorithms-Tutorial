## 🤔 什么是哈希表？

### 哈希表的工作原理

```
哈希表 = 哈希函数 + 数组

1. 哈希函数：把 Key 转换成数组索引
2. 数组：存储 Value

例如：
Key: "apple" → 哈希函数 → 索引 3 → 数组[3] = "苹果"
Key: "banana" → 哈希函数 → 索引 7 → 数组[7] = "香蕉"
```

### 哈希函数的要求

```
好的哈希函数应该：
1. 确定性：同样的 Key 总产生同样的索引
2. 均匀分布：Key 尽量均匀分布在数组中
3. 高效计算：计算速度快
```

### 简单的哈希函数示例

```python
def simple_hash(key, size):
    """
    简单的哈希函数
    把字符串的每个字符的 ASCII 码加起来
    """
    total = 0
    for char in key:
        total += ord(char)
    return total % size


# 测试
size = 10
print(f"apple hash: {simple_hash('apple', size)}")   # (97+112+112+108+101) % 10 = 2
print(f"banana hash: {simple_hash('banana', size)}") # (98+97+110+97+110+97) % 10 = 1
print(f"cherry hash: {simple_hash('cherry', size)}") # (99+104+101+114+114+121) % 10 = 3
```

## 📐 哈希碰撞

### 什么是哈希碰撞？

```
不同的 Key 通过哈希函数得到了相同的索引！

例如：
- hash("apple") = 3
- hash("pear") = 3

两个不同的水果，都映射到索引 3！

这就是"哈希碰撞"
```

### 解决哈希碰撞的方法

#### 方法1：链地址法（Separate Chaining）

```
每个数组位置存一个链表（或其他数据结构）
多个元素映射到同一位置时，用链表串起来

索引 0: [key1, value1] → [key2, value2]
索引 1: [key3, value3]
索引 2: []
索引 3: [key4, value4]
...
```

```python
class HashTableSeparateChaining:
    """哈希表 - 链地址法"""
    def __init__(self, size=10):
        self.size = size
        self.buckets = [[] for _ in range(size)]  # 每个桶是一个链表

    def _hash(self, key):
        """哈希函数"""
        return hash(key) % self.size

    def put(self, key, value):
        """插入/更新"""
        index = self._hash(key)
        bucket = self.buckets[index]

        # 查找是否已存在
        for i, (k, v) in enumerate(bucket):
            if k == key:
                bucket[i] = (key, value)  # 更新
                return

        # 不存在，追加到链表末尾
        bucket.append((key, value))

    def get(self, key):
        """获取值"""
        index = self._hash(key)
        bucket = self.buckets[index]

        for k, v in bucket:
            if k == key:
                return v

        return None  # 没找到

    def delete(self, key):
        """删除"""
        index = self._hash(key)
        bucket = self.buckets[index]

        for i, (k, v) in enumerate(bucket):
            if k == key:
                del bucket[i]
                return True

        return False

    def __str__(self):
        result = []
        for i, bucket in enumerate(self.buckets):
            if bucket:
                result.append(f"{i}: {bucket}")
        return "{" + ", ".join(result) + "}"


# 测试
ht = HashTableSeparateChaining()
ht.put("apple", "苹果")
ht.put("banana", "香蕉")
ht.put("cherry", "樱桃")
ht.put("apple", "红苹果")  # 更新

print(ht)  # {0: [('cherry', '樱桃')], 2: [('apple', '红苹果')], 5: [('banana', '香蕉')]}
print(ht.get("apple"))  # 红苹果
print(ht.get("banana"))  # 香蕉
ht.delete("banana")
print(ht.get("banana"))  # None
```

#### 方法2：开放地址法（Open Addressing）

```
探测策略：
1. 线性探测：hash(key) + 1, hash(key) + 2, ...
2. 二次探测：hash(key) + 1², hash(key) + 2², ...
3. 双重哈希：用另一个哈希函数
```

```python
class HashTableOpenAddressing:
    """哈希表 - 开放地址法（线性探测）"""
    def __init__(self, size=10):
        self.size = size
        self.keys = [None] * size
        self.values = [None] * size

    def _hash(self, key):
        """哈希函数"""
        return hash(key) % self.size

    def put(self, key, value):
        """插入/更新"""
        index = self._hash(key)
        start_index = index

        # 线性探测找空位
        while self.keys[index] is not None:
            if self.keys[index] == key:
                self.values[index] = value  # 更新
                return
            index = (index + 1) % self.size
            if index == start_index:
                raise OverflowError("哈希表已满")

        self.keys[index] = key
        self.values[index] = value

    def get(self, key):
        """获取值"""
        index = self._hash(key)
        start_index = index

        while self.keys[index] is not None:
            if self.keys[index] == key:
                return self.values[index]
            index = (index + 1) % self.size
            if index == start_index:
                break

        return None

    def delete(self, key):
        """删除（标记为已删除）"""
        index = self._hash(key)
        start_index = index

        while self.keys[index] is not None:
            if self.keys[index] == key:
                self.keys[index] = None  # 标记删除
                self.values[index] = None
                return True
            index = (index + 1) % self.size
            if index == start_index:
                break

        return False


# 测试
ht = HashTableOpenAddressing()
ht.put("apple", "苹果")
ht.put("banana", "香蕉")
ht.put("cherry", "樱桃")
print(ht.get("apple"))   # 苹果
print(ht.get("cherry"))  # 樱桃
```

## 💻 Python 内置的哈希表

### 字典（Dict）

```python
# 创建
d = {}  # 空字典
d = {"apple": "苹果", "banana": "香蕉"}

# 插入/更新
d["cherry"] = "樱桃"
d["apple"] = "红苹果"

# 查找
print(d["apple"])  # 红苹果

# 删除
del d["banana"]

# 判断存在
print("apple" in d)  # True

# 遍历
for key, value in d.items():
    print(f"{key}: {value}")

# 方法
d.get("pear", "没找到")  # 没找到（默认值）
d.keys()    # 所有键
d.values()  # 所有值
d.items()   # 所有键值对
```

### 集合（Set）

```python
# 创建
s = {1, 2, 3, 4, 5}

# 添加
s.add(6)
s.update([7, 8])  # 添加多个

# 删除
s.remove(3)   # 不存在会报错
s.discard(3)  # 不存在不报错

# 判断存在

# 集合运算
a = {1, 2, 3, 4}
b = {3, 4, 5, 6}

print(a - b)  # {1, 2} 差集
print(a ^ b)  # {1, 2, 5, 6} 对称差集
```

## 📊 哈希表的性能分析

### 时间复杂度

| 操作 | 平均情况 | 最坏情况 |
|------|---------|---------|
| 插入 | O(1) | O(n) |
| 查找 | O(1) | O(n) |
| 删除 | O(1) | O(n) |

**最坏情况发生在：哈希碰撞严重**

### 负载因子（Load Factor）

```
负载因子 α = 元素数量 / 哈希表大小

当 α 过大时，哈希碰撞增加，性能下降

解决：扩容（Resize/Rehash）
- 当 α > 0.75 时，扩大哈希表
- 新建一个更大的哈希表，重新哈希所有元素

Python 的 dict 会自动扩容！
```

## 🧪 经典面试题

### 题目1：两数之和

```python
def two_sum(nums, target):
    """
    找数组中两个数的和等于 target
    返回这两个数的索引

    哈希表解法 O(n)
    """
    seen = {}  # value -> index

    for i, num in enumerate(nums):
        complement = target - num
        if complement in seen:
            return [seen[complement], i]
        seen[num] = i

    return []


# 测试
nums = [2, 7, 11, 15]
target = 9
print(two_sum(nums, target))  # [0, 1]  因为 nums[0] + nums[1] = 2 + 7 = 9
```

### 题目2：判断字符串是否有重复字符

def has_duplicate(s):
    """判断字符串是否有重复字符"""
    seen = set()
    for char in s:
        if char in seen:
            return True
        seen.add(char)
    return False


# 更简洁的写法
def has_duplicate_v2(s):
    return len(s) != len(set(s))


# 测试

print(has_duplicate("abcaefg"))  # True

### 题目3：找出第一个只出现一次的字符

```python
def first_unique_char(s):
    """
    找出第一个只出现一次的字符
    例如："abracadabra" -> 'c'
    """
    from collections import Counter

    count = Counter(s)

    for i, char in enumerate(s):
        if count[char] == 1:
            return i

    return -1


# 测试
print(first_unique_char("abracadabra"))  # 4 ('c')
print(first_unique_char("aabbcc"))       # -1
```

### 题目4：两数之和 II（有序数组）

```python
def two_sum_ii(numbers, target):
    """
    有序数组，找两个数的和等于 target
    返回 1-indexed 的索引
    """
    left, right = 0, len(numbers) - 1

    while left < right:
        total = numbers[left] + numbers[right]
        if total == target:
            return [left + 1, right + 1]
        elif total < target:
            left += 1
        else:
            right -= 1

    return []


# 测试
numbers = [2, 7, 11, 15]
target = 9
print(two_sum_ii(numbers, target))  # [1, 2]  numbers[0] + numbers[1] = 2 + 7 = 9
```

## 📐 哈希表的实际应用

### 1. 缓存（Cache）

```python
class Cache:
    """简单的 LRU 缓存"""
    def __init__(self, capacity):
        self.capacity = capacity
        self.cache = {}
        self.order = []

    def get(self, key):
        if key in self.cache:
            # 移动到末尾（最近使用）
            self.order.remove(key)
            self.order.append(key)
            return self.cache[key]
        return None

    def put(self, key, value):
        if key in self.cache:
            self.order.remove(key)
        elif len(self.cache) >= self.capacity:
            # 删除最久未使用的
            oldest = self.order.pop(0)
            del self.cache[oldest]

        self.cache[key] = value
        self.order.append(key)


# 测试
cache = Cache(3)
cache.put("a", 1)
cache.put("b", 2)
cache.put("c", 3)
print(cache.get("a"))  # 1
cache.put("d", 4)  # 删除 "b"（最久未使用）
print(cache.get("b"))  # None
```

### 2. 排行榜

```python
class Leaderboard:
    """游戏排行榜"""
    def __init__(self):
        self.scores = {}

    def add_score(self, player_id, score):
        if player_id in self.scores:
            self.scores[player_id] += score
        else:
            self.scores[player_id] = score

    def top(self, k):
        """获取前 K 名"""
        # 排序
        sorted_scores = sorted(self.scores.values(), reverse=True)
        return sum(sorted_scores[:k])


# 测试
lb = Leaderboard()
lb.add_score("Alice", 100)
lb.add_score("Bob", 200)
lb.add_score("Charlie", 150)
lb.add_score("Alice", 50)  # Alice 总分变成 150
print(lb.top(2))  # 350 (Bob: 200 + Alice: 150)
```
